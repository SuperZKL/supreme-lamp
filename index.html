2 Set Up Backend for Form Processing

Choose one of these backend solutions:

Option A: Node.js with Express

// Install required packages
npm install express nodemailer cors

// Create server.js
const express = require('express');
const nodemailer = require('nodemailer');
const cors = require('cors');
const app = express();

app.use(cors());
app.use(express.json());

app.post('/api/contact', async (req, res) => {
  const { name, email, subject, message } = req.body;
  
  // Create transporter
  let transporter = nodemailer.createTransporter({
    service: 'gmail',
    auth: {
      user: 'your-email@gmail.com',
      pass: 'your-app-password'
    }
  });
  
  // Email options
  let mailOptions = {
    from: email,
    to: 'your-email@gmail.com',
    subject: `Portfolio Contact: ${subject}`,
    html: `<p>Name: ${name}</p>
           <p>Email: ${email}</p>
           <p>Message: ${message}</p>`
  };
  
  // Send email
  try {
    await transporter.sendMail(mailOptions);
    res.status(200).json({ message: 'Email sent successfully' });
  } catch (error) {
    res.status(500).json({ error: 'Error sending email' });
  }
});

app.listen(3000, () => console.log('Server running on port 3000'));

Option B: PHP Backend

// Create contact.php
<?php
header('Content-Type: application/json');

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $name = $_POST['name'];
    $email = $_POST['email'];
    $subject = $_POST['subject'];
    $message = $_POST['message'];
    
    $to = 'your-email@example.com';
    $headers = "From: $email\r\n";
    $headers .= "Reply-To: $email\r\n";
    $headers .= "Content-Type: text/html; charset=UTF-8\r\n";
    
    $email_body = "<h2>New Contact Form Submission</h2>
                  <p><strong>Name:</strong> $name</p>
                  <p><strong>Email:</strong> $email</p>
                  <p><strong>Subject:</strong> $subject</p>
                  <p><strong>Message:</strong> $message</p>";
    
    if (mail($to, $subject, $email_body, $headers)) {
        echo json_encode(['message' => 'Email sent successfully']);
    } else {
        echo json_encode(['error' => 'Failed to send email']);
    }
}
?>

3 Integrate with a CMS

Connect your portfolio to a headless CMS for easy content management:

Option A: WordPress REST API

// Fetch blog posts from WordPress
async function fetchBlogPosts() {
  try {
    const response = await fetch('https://your-wordpress-site.com/wp-json/wp/v2/posts');
    const posts = await response.json();
    
    // Display posts in your portfolio
    posts.forEach(post => {
      console.log(post.title.rendered);
      // Add to your blog section
    });
  } catch (error) {

  }
}

Add event tracking for user interactions:

// Example: Track button clicks
document.querySelectorAll('.btn').forEach(button => {
  button.addEventListener('click', () => {
    gtag('event', 'button_click', {
      'event_category': 'engagement',
      'event_label': button.textContent
    });
  });
});

5 Optimize Images

Optimize your images for faster loading:

  • Use WebP format for better compression
  • Resize images to appropriate dimensions
  • Implement lazy loading
  • Use responsive images with srcset
// Example of responsive images
<img 
  src="image-small.jpg" 
  srcset="image-small.jpg 500w, image-medium.jpg 1000w, image-large.jpg 2000w"
  sizes="(max-width: 768px) 100vw, 50vw"
  alt="Project image"
  loading="lazy"
>
